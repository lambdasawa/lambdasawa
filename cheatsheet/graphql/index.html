gq -i https://api.spacex.land/graphql

gq https://api.spacex.land/graphql/ -H 'Content-Type: application/json' \
  -q 'query Foo { dragons { id } }'

gq https://api.spacex.land/graphql/ -H 'Content-Type: application/json' \
  -q 'query Foo($dragonId: ID!) { dragon(id: $dragonId) { id name } }' \
  -v 'dragonId=dragon2'

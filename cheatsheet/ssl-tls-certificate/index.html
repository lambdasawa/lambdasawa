CSRと秘密鍵を作成する

openssl req -new -newkey rsa:2048 -nodes -out foo_example_com.csr -keyout foo_example_com.key -subj "/C=JP/ST=hoge/L=buzz/O=foo/OU=fizz/CN=foo.example.com"
